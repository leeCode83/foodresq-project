buyers
├── id (PK, VARCHAR)
├── email (UNIQUE, VARCHAR)
├── password (VARCHAR)
├── name (VARCHAR)
├── phone (VARCHAR, NULLABLE)
├── address (TEXT, NULLABLE)
├── latitude (DECIMAL(10,8), NULLABLE)
├── longitude (DECIMAL(11,8), NULLABLE)
├── created_at (TIMESTAMP)
└── updated_at (TIMESTAMP)

sellers
├── id (PK, VARCHAR)
├── email (UNIQUE, VARCHAR)
├── password (VARCHAR)
├── phone (VARCHAR, NULLABLE)
├── profile_image (VARCHAR, NULLABLE)
├── business_name (VARCHAR)
├── business_type (ENUM: BAKERY, RESTAURANT, CAFE, GROCERY, CATERING, DESSERT, FAST_FOOD, OTHER)
├── description (TEXT, NULLABLE)
├── address (TEXT)
├── latitude (DECIMAL(10,8))
├── longitude (DECIMAL(11,8))
├── operating_hours (JSON)
├── rating (DECIMAL(3,2), DEFAULT 0)
├── total_reviews (INT, DEFAULT 0)
├── total_orders (INT, DEFAULT 0)
├── created_at (TIMESTAMP)
└── updated_at (TIMESTAMP)

foods
├── id (PK, VARCHAR)
├── seller_id (FK → sellers.id)
├── title (VARCHAR)
├── description (TEXT)
├── category (ENUM: BAKERY, RESTAURANT, CAFE, GROCERY, CATERING, DESSERT, FAST_FOOD, OTHER)
├── image_url (VARCHAR)
├── original_price (DECIMAL(10,2))
├── discounted_price (DECIMAL(10,2))
├── discount_percentage (INT)
├── quantity (INT)
├── available_quantity (INT)
├── pickup_time_start (TIMESTAMP)
├── pickup_time_end (TIMESTAMP)
├── is_active (BOOLEAN, DEFAULT true)
├── created_at (TIMESTAMP)
└── updated_at (TIMESTAMP)

transactions
├── id (PK, VARCHAR)
├── order_number (UNIQUE, VARCHAR)
├── buyer_id (FK → buyers.id)
├── seller_id (FK → sellers.id)
├── food_id (FK → foods.id)
├── quantity (INT)
├── total_price (DECIMAL(10,2))
├── status (ENUM: PENDING, PAID, READY_FOR_PICKUP, COMPLETED, CANCELLED)
├── created_at (TIMESTAMP)
├── paid_at (TIMESTAMP, NULLABLE)
├── completed_at (TIMESTAMP, NULLABLE)
├── cancelled_at (TIMESTAMP, NULLABLE)
├── cancellation_reason (TEXT, NULLABLE)
├── payment_method (ENUM: CASH, TRANSFER, E_WALLET, CREDIT_CARD)
├── payment_id (VARCHAR, NULLABLE)
├── payment_proof (VARCHAR, NULLABLE)
├── payment_status (ENUM: pending, success, failed)
├── pickup_code (VARCHAR, NULLABLE)
├── pickup_time (TIMESTAMP)
├── pickup_verified_at (TIMESTAMP, NULLABLE)
└── updated_at (TIMESTAMP)

reviews
├── id (PK, VARCHAR)
├── transaction_id (FK → transactions.id, UNIQUE)
├── buyer_id (FK → buyers.id)
├── seller_id (FK → sellers.id)
├── food_id (FK → foods.id)
├── rating (INT)
├── comment (TEXT, NULLABLE)
├── food_quality (INT, NULLABLE)
├── service_quality (INT, NULLABLE)
├── value_for_money (INT, NULLABLE)
├── created_at (TIMESTAMP)
└── updated_at (TIMESTAMP)
